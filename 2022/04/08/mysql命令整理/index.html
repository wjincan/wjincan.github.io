1.数据库和表的基本操作

创建数据库

CREATE DATABASE demo;

查看创建数据库的信息

SHOW CREATE DATABASE Demo;

使用该数据库

USE demo;

将数据库格式从gbk修改为gbk_bin

ALTER DATABASE demo DEFAULT CHARACTER SET gbk COLLATE gbk_bin;

删除数据库

DROP DATABASE demo;

查看所有数据库

SHOW DATABASES;

创建表，并建三个列

CREATE TABLE stu(
stu_id INT(10),
stu_name VARCHAR(20),
stu_age INT(10)
);

查看列信息

SHOW CREATE TABLE stu;

将表demo改为demostu

ALTER TABLE stu RENAME demostu;

将stu_age改为 stu_sex VARCHAR(10)

ALTER TABLE demostu CHANGE stu_age stu_sex VARCHAR(10);

stu_sex字段类型改为char

ALTER TABLE demostu MODIFY stu_sex CHAR;

增加字段

ALTER TABLE demostu ADD stu_age INT(10);

删除字段

ALTER TABLE demostu DROP stu_age;

字段stu_name放在stu_sex后面

ALTER TABLE demostu MODIFY stu_name VARCHAR(20) AFTER stu_sex;

stu_name放在第一行

ALTER TABLE demostu MODIFY stu_name VARCHAR(20) FIRST;

删除表

DROP TABLE demostu;

查全表

SELECT * FROM stu;

查指定字段

SELECT sid,sname FROM stu;

带关系运算符的查询 where

SELECT * FROM stu WHERE gender=’female’;

带and关键字的查询

SELECT * FROM stu WHERE age>19 AND gender=’female’;

带or关键字的查询

SELECT * FROM stu WHERE age>19 OR gender=’female’;
SELECT * FROM stu WHERE (sid=’10001’ OR age>19 )AND gender=’female’;

带 is null 或 is not null关键字

SELECT *FROM stu WHERE age IS NULL;

between and

SELECT * FROM stu WHERE age BETWEEN 19 AND 21;
SELECT * FROM stu WHERE age NOT BETWEEN 19 AND 21;

like 模糊查询

SELECT FROM stu WHERE sname LIKE ‘_____’; #5个下划线代表5个字母，查询5个字母的名字
SELECT* FROM stu WHERE sname LIKE ‘___i’; # 查询第四个字母为i
SELECT FROM stu WHERE sname LIKE ‘z%’ ; #z开头， %代表任意字符
SELECT* FROM stu WHERE sname LIKE ‘%a%’ ; #中间包含a %代表任意字符

distinct 去重数据

SELECT DISTINCT sal FROM emp;

排序查询

SELECT * FROM emp ORDER BY sal ASC; #asc升序排序
SELECT * FROM emp ORDER BY sal DESC; #desc降序排序
SELECT * FROM emp ORDER BY sal ASC,empno ASC; #工资升序排序，工资相同员工号升序

#聚合函数

COUNT(*|1|列名) 返回行数

SELECT COUNT() FROM emp;
SELECT COUNT(1) AS totle FROM emp; #as可省略 设置列名的别名 totle
SELECT COUNT() AS total FROM emp WHERE sal>2000;
SELECT COUNT(comm),COUNT(mgr) FROM emp ;

sum() 求和

SELECT SUM(sal) FROM emp;
SELECT SUM(sal+IFNULL(comm,0)) AS total FROM emp; #奖金和工资 ifnull()函数用0替代null

avg() 平均

SELECT AVG(sal) FROM emp;

max() 最大值

SELECT MAX(sal) FROM emp;

min() 最小值

SELECT MIN(sal) FROM emp;

分组查询

SELECT deptno,SUM(sal) FROM emp GROUP BY deptno; #按照部门查询工资总和
SELECT deptno,COUNT() FROM emp GROUP BY deptno; #按照部门查询人数
SELECT deptno,COUNT() FROM emp WHERE sal>1000 GROUP BY deptno; #按照部门查询工资大于1000的人数

having字句 对于复杂语句，查询完成后再过滤一次

SELECT deptno,SUM(sal) FROM emp GROUP BY deptno HAVING SUM(sal)>4000;

limit 分页

SELECT * FROM stu LIMIT 1,3;
SELECT * FROM stu LIMIT 3;#从0开始可以省略
